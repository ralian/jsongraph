cmake_minimum_required(VERSION 3.24)
project(jsongraph LANGUAGES CXX)
include(FetchContent)
include(util.cmake)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

find_package(OpenGL REQUIRED)

set(glfw_SOURCE_DIR glfw)
FetchContent_Declare(
    glfw
    GIT_REPOSITORY  https://github.com/glfw/glfw.git
    GIT_TAG         3.4
)

set(imgui_SOURCE_DIR imgui)
FetchContent_Declare(
    imgui
    GIT_REPOSITORY  https://github.com/ocornut/imgui.git
    GIT_TAG         v1.92.1
)

set(rapidjson_SOURCE_DIR rapidjson)
FetchContent_Declare(
    rapidjson EXCLUDE_FROM_ALL SYSTEM
    GIT_REPOSITORY  https://github.com/Tencent/rapidjson.git
    GIT_TAG         v1.1.0
)

FetchContent_MakeAvailable(glfw imgui rapidjson)

add_executable(jsongraph
    jsongraph.cpp
    jsongraph_parse.cpp
    main.cpp
    ${imgui_SOURCE_DIR}/imgui_demo.cpp
    ${imgui_SOURCE_DIR}/imgui_draw.cpp
    ${imgui_SOURCE_DIR}/imgui_tables.cpp
    ${imgui_SOURCE_DIR}/imgui_widgets.cpp
    ${imgui_SOURCE_DIR}/imgui.cpp
    ${imgui_SOURCE_DIR}/backends/imgui_impl_opengl3.cpp
    ${imgui_SOURCE_DIR}/backends/imgui_impl_glfw.cpp
)
target_link_libraries(jsongraph PRIVATE glfw OpenGL::GL)

target_include_directories(jsongraph PRIVATE SYSTEM ${imgui_SOURCE_DIR} ${rapidjson_SOURCE_DIR}/include)
